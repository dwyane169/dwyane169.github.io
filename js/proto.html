# JS原型继承方式

# 扩展原型对象

function Person() {}

var p1 = new Person();

Person.prototype.run = function() {
    console.log("I'm running");
};

console.log(p1.run);

//此时，p1 是可以访问到 run 方法的，我们就说 p1 继承自 Person.prototype 。

# 替换原型对象

function Person() {}

// 替换原型对象
Person.prototype = {
    constructor: Person,  // 重点
    run: function() {},
    say: function() {},
    sing: function() {},
    walk: function() {}
};

// 实例可以访问
var p1 = new Person();
p1.run;
p1.say;

# 经典的面试题

function Person() {}

Person.prototype.run = function() {
    // code...
};

var p1 = new Person();   // p1.__proto__ 指向默认的原型对象

Person.prototype = {
    constructor: Person,
    say: function() {
        // code...
    }
};

var p2 = new Person();   // p2.__proto__ 指向新的原型对象

console.log(typeof p1.say); // undefined
console.log(typeof p2.say); // "function"

# 原型式继承(也叫经典继承)

// 让 obj4 继承自 obj3
// obj4.__proto__ == obj3;
// 但是在这里我们不能直接使用 obj4.__proto__ == obj3 ，因为 __proto__ 属性时非标准属性，有浏览器兼容问题
var obj3 = { name:'ls', age: 18 };

function F() {}

F.prototype = obj3;  // 让 F 的实例可以访问到 obj3 的属性和方法

var obj4 = new F();

obj4.name = 'ww';

obj4.gender = '女';

console.log(obj4.name);  // obj4 有自己的 name，打印 'ww'

console.log(obj4.gender); // obj4 有自己的 gender，打印 '女'

console.log(obj4.age);  // obj4 没有自己的 age，访问 obj3 中的 age，打印 18

# ES6 中如何获取以及改变对象的原型

// 定义一个父类 person
let person = {
    say() {
        console.log('Hi');
    }
};

// ES5 中就有的 Object.getPrototypeOf()
let p1 = Object.create(person);
Object.getPrototypeOf(p1) === person; // true

// ES6 中我们可以设置实例的原型了
// 通过 Object.setPrototypeOf()
// 两个参数：
// 1、要改变原型的对象
// 2、替代原型的对象
let animal = {
    say() {
        console.log('I can not speak');
    }
};

Object.setPrototypeOf(p1, animal);

Object.getPrototypeOf(p1) === person; // false
Object.getPrototypeOf(p1) === animal; // true

# super 对象

TIP

super 的引入主要是为了我们更方便地访问对象的原型。

// A 类
let A = {
    say() {
        return 'Hello';
    }
};

// B 类
let B = {
    say() {
        return super.say() + ', World';
    }
}

// A类 是 B类的父类
Object.setPrototypeOf(B, A);

B.say(); // "Hello, World"

# ES6函数值得注意的点

# 形参

// 你可以把第二个参数的默认值设置为第一个参数，但是不能把第一个参数的默认值设置为第二个参数。
function getAge(age=10, newAge=age) {
	console.log(newAge);
}
getAge(); // 10

//另一方面，函数的默认值还可以是一个表达式

function getValue() {
    return 3;
}

function add(first, second=getValue()) {
    return first + second;
}

add(1, 3); // 4
add(2); // 5
